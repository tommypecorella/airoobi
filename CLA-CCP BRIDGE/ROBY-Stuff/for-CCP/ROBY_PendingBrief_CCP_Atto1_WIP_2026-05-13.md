```sql
CREATE TABLE evalobi (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  object_title TEXT NOT NULL,

  -- Versioning (re-evaluation support)
  version INTEGER NOT NULL DEFAULT 1,
  supersedes_evalobi_id UUID REFERENCES evalobi(id),  -- old version pointer

  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_evalobi_owner ON evalobi(current_owner_id);
CREATE INDEX idx_evalobi_original_seller ON evalobi(original_seller_id);
CREATE INDEX idx_evalobi_outcome ON evalobi(evaluation_outcome);

-- Storico append-only per re-evaluation chain
CREATE TABLE evalobi_history (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  event_metadata JSONB,
  prev_owner_id UUID REFERENCES profiles(id),

```

```sql
-- Materialized view per metriche aggregate (refresh ogni 5 min via pg_cron)
CREATE MATERIALIZED VIEW sla_metrics_30d AS
SELECT
  COUNT(*) AS total_requests_30d,
  AVG(EXTRACT(EPOCH FROM (decided_at - created_at)) / 3600) AS avg_response_hours,
  PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY EXTRACT(EPOCH FROM (decided_at - created_at)) / 3600) AS median_response_hours,
  (COUNT(*) FILTER (WHERE EXTRACT(EPOCH FROM (decided_at - created_at)) / 3600 <= 48)) * 100.0 / COUNT(*) AS pct_under_48h,
  (COUNT(*) FILTER (WHERE EXTRACT(EPOCH FROM (decided_at - created_at)) / 3600 <= 24)) * 100.0 / COUNT(*) AS pct_under_24h,
  -- Distribution buckets
  COUNT(*) FILTER (WHERE EXTRACT(EPOCH FROM (decided_at - created_at)) / 3600 <= 12) AS bucket_0_12h,
  COUNT(*) FILTER (WHERE EXTRACT(EPOCH FROM (decided_at - created_at)) / 3600 > 12 AND EXTRACT(EPOCH FROM (decided_at - created_at)) / 3600 <= 24) AS bucket_12_24h,
  COUNT(*) FILTER (WHERE EXTRACT(EPOCH FROM (decided_at - created_at)) / 3600 > 24 AND EXTRACT(EPOCH FROM (decided_at - created_at)) / 3600 <= 36) AS bucket_24_36h,
  COUNT(*) FILTER (WHERE EXTRACT(EPOCH FROM (decided_at - created_at)) / 3600 > 36 AND EXTRACT(EPOCH FROM (decided_at - created_at)) / 3600 <= 48) AS bucket_36_48h,
  COUNT(*) FILTER (WHERE EXTRACT(EPOCH FROM (decided_at - created_at)) / 3600 > 48) AS bucket_over_48h,
  NOW() AS refreshed_at
FROM evaluation_requests
WHERE decided_at IS NOT NULL
  AND created_at >= NOW() - INTERVAL '30 days';

CREATE UNIQUE INDEX idx_sla_metrics_30d ON sla_metrics_30d(refreshed_at);

-- Refresh schedule
SELECT cron.schedule('refresh-sla-metrics-30d', '*/5 * * * *', 'REFRESH MATERIALIZED VIEW CONCURRENTLY sla_metrics_30d');
```

CREATE TABLE email_send_log (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES profiles(id),

);

-- View daily count for cap check

SELECT COUNT(*) FILTER (WHERE send_status = 'sent') AS sent_today
